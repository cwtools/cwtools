#selecting target relative
letter_event = {
	id = GBC.1
	desc = EVTDESC_GBC_1
	
	min_age = 16
	has_global_flag = black_death_happened

	trigger = {
		has_dlc = "Game Book"
		NOT = { has_character_flag = gbc_mourning }
		OR = {
			any_friend = {
				is_adult = yes
				ai = yes
				trait = has_bubonic_plague
			}			
	    	any_dynasty_member = {
				is_adult = yes
				ai = yes
				trait = has_bubonic_plague
			}
		}
   }

	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				any_friend = {
					is_adult = yes
					ai = yes
					trait = has_bubonic_plague
				}
			}
			random_friend = {
				limit = {
					is_adult = yes
					ai = yes
					trait = has_bubonic_plague        
				}
				save_event_target_as = GBC_dead_one               
			}
		}
		if = {
			limit = {
				NOT = {
					any_friend = {
						is_adult = yes
						ai = yes
						trait = has_bubonic_plague
					}
				}
			}
			random_dynasty_member = {
				limit = {
					is_adult = yes
					ai = yes
					trait = has_bubonic_plague        
				}
				save_event_target_as = GBC_dead_one               
			}
		}
	}

	#Kill plagued char
	option = {
	name = EVTOPTA_GBC_3

		event_target:GBC_dead_one = {
			add_trait = falconer
			death = { death_reason = death_trait }
		}
		set_character_flag = gbc_mourning
		character_event = { id = GBC.2 days = 10 }
	}
}

#Trying to find a lover
character_event = {
	id = GBC.100
	
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		ai = no
	}

	immediate = {

		#Rival: liege
		if = {
			limit = { liege = { ai = yes is_married = yes block_general_event_trigger = no } }

			liege = {
				random_spouse = {
					limit = {
						block_general_event_trigger = no
						OR = {
							AND = {
								is_female = yes
								ROOT = { prefers_men_trigger = no }
							}
							AND = {
								is_female = no
								ROOT = { prefers_men_trigger = yes }
							}
						}
						NOT = { is_close_relative = ROOT }
					}
					save_event_target_as = gbc_lover
					break = yes
				}
			}
		}

		#Rival: someone else under the same liege
		if = {
			limit = { liege = { any_vassal = { NOR = { character = ROOT ai = no is_married = no block_general_event_trigger = no } } } }
				
			#Find a lover
			liege = {

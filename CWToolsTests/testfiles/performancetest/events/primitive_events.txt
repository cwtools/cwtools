country_event = {
	id = primitive.10
	hide_window = yes

	trigger = {
		is_country_type = primitive
		has_country_flag = industrial_age
		NOT = { has_country_flag = recently_advanced }
	}

	mean_time_to_happen = {
		years = 100
	}

	immediate = {
		set_timed_country_flag = {
			flag = recently_advanced
			days = 2160 #60yrs
		}
		remove_country_flag = industrial_age
		set_country_flag = machine_age
		set_primitive_age = machine_age
		home_planet = {
			create_army = {
				name = "NAME_Industrial_Army"
				owner = root
				species = root
				type = "industrial_army"
			}
			random_tile = {
				limit = {
					has_blocker = no
					has_building = no
					has_grown_pop = no
				}
				set_building = "building_primitive_factory"
				create_pop = {
					species = owner
				}
				clear_deposits = yes
				add_deposit = d_mineral_deposit
			}
			observation_outpost_owner = {
				country_event = { id = primitive.11 }
			}
		}
	}
}

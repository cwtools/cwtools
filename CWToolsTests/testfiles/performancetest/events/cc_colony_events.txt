	}
	picture = GFX_evt_cc_military_ceremony
	
	is_triggered_only = yes
	location = root
	
	immediate = {
		owner = {
			if = {
				limit = { NOT = { has_country_flag = cc_has_had_militarist_story } }
				set_country_flag = cc_has_had_militarist_story
			}
			set_timed_country_flag = {
				flag = cc_recent_militarist_story
				days = 3600
			}
		}
			
	
		if = {
			limit = {
				any_owned_pop = { is_same_species = owner }
			}
			owner = {
				create_leader = {
					type = general
					species = root.owner_species
					name = random
					skill = 0
				}
			}
			else = {
				random_owned_pop = {
					limit = { species = { pops_have_happiness = yes } }
					save_event_target_as = cc_leader_pop
				}
				owner = {
					create_leader = {
						type = general
						species = cc_leader_pop.species
						name = random
						skill = 0
					}
				}
			}
		}
		last_created_leader = {
			save_event_target_as = cc_colony_leader
			exile_leader_as = cc_militarist_colony_leader
		}
	}

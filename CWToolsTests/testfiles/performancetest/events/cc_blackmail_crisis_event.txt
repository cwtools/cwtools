country_event = {
	id = cc_events_blackmail.201
	title = cc_events_blackmail.201.name
	desc = cc_events_blackmail.201.desc
	picture = GFX_evt_news_room
	
	is_triggered_only = yes
	
	immediate = {
		random_owned_leader = {
			limit = { NOT = { leader_class = ruler } }
			save_event_target_as = cc_first_corrupt
		}
		random_owned_leader = {
			limit = {
				NOR = {
					leader_class = ruler
					is_same_value = event_target:cc_first_corrupt
				}
			}
			save_event_target_as = cc_second_corrupt
		}
		random_owned_leader = {
			limit = {
				NOR = {
					leader_class = ruler
					is_same_value = event_target:cc_first_corrupt
					is_same_value = event_target:cc_second_corrupt
				}
			}
			save_event_target_as = cc_third_corrupt
		}
		if = {
			limit = {
				any_owned_leader = {
					NOR = {
						leader_class = ruler
						is_same_value = event_target:cc_first_corrupt
						is_same_value = event_target:cc_second_corrupt
						is_same_value = event_target:cc_third_corrupt
					}
				}
			}
			random_owned_leader = {
				limit = {
					NOR = {
						leader_class = ruler
						is_same_value = event_target:cc_first_corrupt
						is_same_value = event_target:cc_second_corrupt
						is_same_value = event_target:cc_third_corrupt
					}
				}
				save_event_target_as = cc_fourth_corrupt
			}
		}
	}
			
	
	option = {
		name = cc_events_blackmail.201.A
		
		custom_tooltip = cc_events_blackmail.201.A.tooltip
		
		hidden_effect = {
			event_target:cc_first_corrupt = { kill_leader = { show_notification = no } }
			event_target:cc_second_corrupt = { kill_leader = { show_notification = no } }
			event_target:cc_third_corrupt = { kill_leader = { show_notification = no } }
		}
		if = {
			limit = { exists = event_target:cc_fourth_corrupt }
			custom_tooltip = cc_events_blackmail.201.A.tooltip2
			hidden_effect = { event_target:cc_fourth_corrupt = { kill_leader = { show_notification = no } } }
		}
		
		ai_chance = {
			factor = 2.5
			modifier = {
				factor = 0.5
				influence < 200
			}
			modifier = {
				factor = 0.8
				if = {
					limit = { exists = event_target:cc_first_corrupt }
					event_target:cc_first_corrupt = { has_skill > 4 }
				}
			}
			modifier = {
				factor = 0.8
				if = {
					limit = { exists = event_target:cc_second_corrupt }
					event_target:cc_second_corrupt = { has_skill > 4 }
				}
			}
			modifier = {
				factor = 0.8
				if = {
					limit = { exists = event_target:cc_third_corrupt }
					event_target:cc_third_corrupt = { has_skill > 4 }
				}
			}
			modifier = {
				factor = 0.8
				if = {
					limit = { exists = event_target:cc_fourth_corrupt }
					event_target:cc_fourth_corrupt = { has_skill > 4 }
				}
			}
		}
	}
	option = {
		name = cc_events_blackmail.201.B
		
		add_modifier = {
			modifier = cc_outrage_at_corruption
			years = 10
		}
		
		ai_chance = {
			modifier = {
				factor = 2
				NOT = { any_owned_planet = { unrest > 0 } }
			}
			modifier = {
				factor = 0.5
				influence > 300
			}
			modifier = {
				factor = 0.5
				AND = {
					if = {
						limit = { exists = event_target:cc_first_corrupt }
						event_target:cc_first_corrupt = { has_skill < 4 }
					}
					if = {
						limit = { exists = event_target:cc_second_corrupt }
						event_target:cc_second_corrupt = { has_skill < 4 }
					}
					if = {
						limit = { exists = event_target:cc_third_corrupt }
						event_target:cc_third_corrupt = { has_skill < 4 }
					}
					if = {
						limit = { exists = event_target:cc_fourth_corrupt }
						event_target:cc_fourth_corrupt = { has_skill < 4 }
					}
				}
			}
			modifier = {
				factor = 1.2
				if = {
					limit = { exists = event_target:cc_first_corrupt }
					event_target:cc_first_corrupt = { has_skill > 5 }

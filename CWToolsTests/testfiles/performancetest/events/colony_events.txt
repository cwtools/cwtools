### Migrating Forests
planet_event = {
	id = colony.1
	title = "colony.1.name"
	desc = "colony.1.desc"
	picture = GFX_evt_alien_nature
	show_sound = event_alien_nature
	location = ROOT
	
	trigger = {
		has_owner = yes
		original_owner = yes
		is_homeworld = no
		owner = { 
			is_ai = no
			NOT = { has_authority = auth_machine_intelligence }
		}
		has_ground_combat = no
		is_occupied_flag = no
		num_pops > 0
		is_capital = no
		OR = {
			is_planet_class = pc_continental
			is_planet_class = pc_tropical
			is_planet_class = pc_arid
			is_planet_class = pc_tundra
			is_planet_class = pc_savannah
		}
		NOT = { 
			has_global_flag = migrating_forests_global
			has_planet_flag = colony_event
		}
	}

	is_triggered_only = yes
	#mean_time_to_happen = {
	#	months = 640
	#}
	
	immediate = {
		set_global_flag = migrating_forests_global
		set_planet_flag = colony_event
		random_tile = {
			limit = { 
				has_blocker = no
				has_grown_pop = no
				has_building = no
				has_building_construction = no
			}
			set_blocker = "tb_wandering_forests"
		}
		random_tile = {
			limit = { 
				has_blocker = no
				has_grown_pop = no
				has_building = no
				has_building_construction = no
			}
			set_blocker = "tb_wandering_forests"
		}
		random_tile = {
			limit = { 
				has_blocker = no
				has_grown_pop = no
				has_building = no
				has_building_construction = no
			}
			set_blocker = "tb_wandering_forests"
		}
		add_modifier = {
			modifier = "migrating_forests"
			days = -1
		}
	}
	
	option = {
		name = colony.1.a
		begin_event_chain = {
			event_chain = "migrating_forests_chain"
			target = ROOT
		}
		enable_special_project = {
			name = "MIGRATING_FORESTS_1_PROJECT"
			location = this
			owner = root
		}
		enable_special_project = {
			name = "MIGRATING_FORESTS_2_PROJECT"
			location = this
			owner = root
		}
	}
}

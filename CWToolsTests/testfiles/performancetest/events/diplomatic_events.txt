country_event = {
	id = diplomatic.9

	hide_window = yes
	
	trigger = {
		#
		always = no
		#
		any_owned_ship = {
			is_ship_class = shipclass_science_ship
			exists = solar_system
			solar_system = {
				exists = space_owner
				space_owner = {
					has_communications = root
					NOR = {
						is_country = ROOT
						is_at_war_with = ROOT
						is_subject = yes
					}
					OR = {
						has_ethic = ethic_materialist
						has_ethic = ethic_fanatic_materialist
						AND = {
							ROOT = {
								NOT = { has_modifier = alien_genome_research }
								has_technology = tech_society_genome_447
							}
							OR = {
								has_ethic = ethic_xenophobe
								has_ethic = ethic_fanatic_xenophobe
							}
						}
						AND = {
							#exists = leader
							OR = {
								has_ethic = ethic_spiritualist
								has_ethic = ethic_fanatic_spiritualist
							}
							ROOT = {
								NOR = {
									has_ethic = ethic_spiritualist
									has_ethic = ethic_fanatic_spiritualist
								}
							}
						}
					}
				}
			}
		}
	}
	
	#is_triggered_only = yes
	mean_time_to_happen = { months = 3000 }
	
	immediate = {
		random_owned_ship = {
			limit = {
				is_ship_class = shipclass_science_ship
				exists = solar_system
				solar_system = {
					exists = space_owner
					space_owner = {
						has_communications = root
						NOR = {
							is_country = ROOT
							is_at_war_with = ROOT
							is_subject = yes
						}
						OR = {
							has_ethic = ethic_materialist
							has_ethic = ethic_fanatic_materialist
							AND = {
								ROOT = {
									NOT = { has_modifier = alien_genome_research }
									has_technology = tech_society_genome_447
								}
								OR = {
									has_ethic = ethic_xenophobe
									has_ethic = ethic_fanatic_xenophobe
								}
							}
							AND = {
								#exists = leader
								OR = {
									has_ethic = ethic_spiritualist
									has_ethic = ethic_fanatic_spiritualist
								}
								ROOT = {
									NOR = {
										has_ethic = ethic_spiritualist
										has_ethic = ethic_fanatic_spiritualist
									}
								}
							}
						}
					}
				}
			}
			save_event_target_as = encroaching_research_fleet
			if = {
				limit = { exists = leader }
				leader = { save_event_target_as = encroaching_research_leader }
			}
			solar_system = {
				space_owner = {
					switch = {
						trigger = has_ethic
						ethic_materialist = { country_event = { id = diplomatic.10 } }
						ethic_fanatic_materialist = { country_event = { id = diplomatic.10 } }
						ethic_xenophobe = { country_event = { id = diplomatic.12 } }
						ethic_fanatic_xenophobe = { country_event = { id = diplomatic.12 } }
						ethic_spiritualist = { country_event = { id = diplomatic.15 } }
						ethic_fanatic_spiritualist = { country_event = { id = diplomatic.15 } }
					}
				}
			}
		}
	}
}

# Encroaching Colonies
planet_event = {
	id = diplomatic.21
	title = "diplomatic.21.name"
	desc = "diplomatic.21.desc"
	picture = GFX_evt_colony_settlement
	location = root

	is_triggered_only = yes

	trigger = {
		NOT = { is_artificial_planet = yes }
		solar_system = {
			NOT = {	exists = space_owner } # colonizing outside borders
			any_neighbor_system = {
				exists = space_owner
				space_owner = {
					is_ai = yes
					is_normal_country = yes
					has_communications = root.owner
					NOR = {
						is_country = root.owner
						is_at_war_with = root.owner
						is_subject = yes
						is_in_federation_with = root.owner
						has_opinion_modifier = { who = root.owner modifier = encroaching_colony }
						opinion = {
							who = root.owner
							value > -100
						}
					}
				}
			}
		}
	}

	immediate = {
		solar_system = {
			random_neighbor_system = {
				limit = {
					exists = space_owner
					space_owner = {
						has_communications = root.owner
						NOT = { is_country = ROOT.owner }
						is_ai = yes
						is_normal_country = yes	   
						NOT = { is_in_federation_with = ROOT.owner }
						NOT = {
							has_opinion_modifier = { who = ROOT.owner modifier = encroaching_colony }
						}
						opinion = {
							who = root.owner
							value > -100
						}
					}
				}
				space_owner = { save_event_target_as = neighbor_contesting_colony }
			}
		}
	}

	diplomatic = yes

	picture_event_data = {
		portrait = event_target:neighbor_contesting_colony
		planet_background = event_target:neighbor_contesting_colony
		graphical_culture = event_target:neighbor_contesting_colony
		city_level = event_target:neighbor_contesting_colony
		room = event_target:neighbor_contesting_colony.ruler
	}
	
	option = {
		name = diplomatic.21.a
		destroy_colony = { keep_buildings = no }
		ai_chance = {
			factor = 0
		}
	}

	option = {
		name = diplomatic.21.b
		event_target:neighbor_contesting_colony = {
			add_opinion_modifier = {
				who = ROOT.owner
				modifier = encroaching_colony
			}
		}
	}
}

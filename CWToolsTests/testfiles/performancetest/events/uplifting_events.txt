### Uplifting Snakeoids
ship_event = {
	id = uplift.6000
	title = "uplift.6000.name"
	desc = "uplift.6000.desc"
	picture = GFX_evt_landing_ship
	location = FROM
	show_sound = event_default

	is_triggered_only = yes

	immediate = {
		owner = { set_country_flag = found_presapients }
		# owner = { country_event = { id = tutorial.57 days = 5 } }
		owner = { country_event = { id = story.4 days = 30 } }
		owner = { set_country_flag = snakeoids }
		FROM = {
			if = {
				limit = { has_modifier = hazardous_weather }
				remove_modifier = hazardous_weather
			}
			if = {
				limit = { has_modifier = unstable_tectonics }
				remove_modifier = unstable_tectonics
			}
			if = {
				limit = { has_modifier = asteroid_impacts }
				remove_modifier = asteroid_impacts
			}
			if = {
				limit = { has_modifier = wild_storms }
				remove_modifier = wild_storms
			}
			if = {
				limit = { has_modifier = low_gravity }
				remove_modifier = low_gravity
			}
			if = {
				limit = { has_modifier = high_gravity }
				remove_modifier = high_gravity
			}
			if = {
				limit = { has_modifier = damaged_ecology }
				remove_modifier = damaged_ecology
			}
			set_planet_flag = snakeoid_planet
			create_species = {
				name = "NAME_Snakeoid"
				class = "PRE_RANDOMTREK"
				portrait = random
				traits = {
					trait = random_presapient_trait
				}
				homeworld = FROM
				sapient = no
			}
			while = {
				limit = {
					count_pops = {
						limit = { is_sapient = no }
						count < 3
					}
					free_pop_tiles > 0
				}
				best_tile_for_pop = {
					create_pop = { species = last_created }
				}
			}
		}
	}

	option = {
		name = uplift.6000.a
		trigger = {
			owner = { has_technology = "tech_epigenetic_triggers" }
		}
	}

	option = {
		name = uplift.6000.b
		trigger = {
			owner = { NOT = { has_technology = "tech_epigenetic_triggers" } }
		}
	}
}

		FROM = {
			if = {
				limit = { has_modifier = hazardous_weather }
				remove_modifier = hazardous_weather
			}
			if = {
				limit = { has_modifier = unstable_tectonics }
				remove_modifier = unstable_tectonics
			}
			if = {
				limit = { has_modifier = asteroid_impacts }
				remove_modifier = asteroid_impacts
			}
			if = {
				limit = { has_modifier = wild_storms }
				remove_modifier = wild_storms
			}
			if = {
				limit = { has_modifier = low_gravity }
				remove_modifier = low_gravity
			}
			if = {
				limit = { has_modifier = high_gravity }
				remove_modifier = high_gravity
			}
			if = {
				limit = { has_modifier = damaged_ecology }
				remove_modifier = damaged_ecology
			}
			create_species = {
				name = random
				class = "PRE_RANDOMTREK"
				portrait = random
				name = "Roachoid"
				traits = {
					trait = trait_presapient_irradiated
					#trait = random_presapient_trait
				}
				homeworld = FROM
				sapient = no
			}
			while = {
				limit = {
					count_pops = {
						limit = { is_sapient = no }
						count < 3
					}
					free_pop_tiles > 0
				}
				best_tile_for_pop = {
					create_pop = { species = last_created }
				}
			}
			set_planet_flag = roachoid_planet
		}
	}
	option = {
		name = uplift.6000.a
		trigger = {
			owner = { has_technology = "tech_epigenetic_triggers" }
		}
	}

	option = {
		name = uplift.6000.b
		trigger = {
			owner = { NOT = { has_technology = "tech_epigenetic_triggers" } }
		}
	}
}

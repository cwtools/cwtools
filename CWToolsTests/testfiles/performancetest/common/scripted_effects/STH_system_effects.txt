	set_country_flag = cleanupRemove
	every_owned_planet = {
		destroy_colony = { keep_buildings = no }
		solar_system = { 
			set_star_flag = remakeSystem
			every_system_planet = { 
				limit = { OR = { is_planet_class = pc_thegreatlink is_planet_class = pc_r_class is_planet_class = pc_unimatrix  } }
				change_pc = pc_r_class
				set_random_rogue_planet_name = yes
			}
			every_system_planet = { limit = { is_star = yes } set_random_star_name = yes  }	
			every_system_planet = {
				limit = { is_star = no }
				remove_planet = yes
			}
			every_system_megastructure = { remove_megastructure = this }
			every_system_ambient_object = { destroy_ambient_object = this }
			rebuild_random_system = yes
			if = {
				limit = { sth_random_maps = yes }
				spawn_random_trek_system = yes
			}
		}
	}
	destroy_country = yes
}

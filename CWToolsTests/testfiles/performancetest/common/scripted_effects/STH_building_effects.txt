#Planet Scope
clear_colony_hubs = {
    every_tile = {
		limit = { 
			OR = { 
				has_building = building_starbase_operations_0
				has_building = building_starbase_operations_1
				has_building = building_starbase_operations_2
				has_building = building_starbase_operations_3
				has_building = helix_ops_0
				has_building = helix_ops_1
				has_building = helix_ops_2
				has_building = helix_ops_3
				has_building = building_colony_shelter
				has_building = building_capital_1
				has_building = building_capital_2
				has_building = building_capital_3
				has_building = building_hab_capital
			} 
		}
		remove_building = yes
	}
}

tile_add_appropriate_mining_building = {
	if = {
		limit = { planet = { is_habitat_planet = yes } }
		set_building = space_building_basic_mine
	}
	if = {
		limit = { planet = { is_starbase_planet = yes } }
		set_building = building_starbase_ore_0
	}
	if = {
		limit = { planet = { is_artificial_planet = no } }
		set_building = building_basic_mine
	}
}

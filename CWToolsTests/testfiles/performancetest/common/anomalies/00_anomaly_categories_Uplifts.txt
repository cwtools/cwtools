# Uplifting Snakeoids
anomaly_category = {
	key = "UPLIFT_SNAKEOID_CAT"
	desc = "UPLIFT_SNAKEOID_CAT_DESC"
	picture = GFX_evt_landing_ship
	level = 1
	#risk = 0
	
	spawn_chance = {
		modifier = {
			add = 5
			is_planet_class = pc_desert
			FROM = { owner = { NOT = { has_country_flag = upl_snakeoid } } }
			FROM = { owner = { is_borg_empire = no is_assimilator = no } }
		}
	}

	on_success = {
		owner = { set_country_flag = upl_snakeoid }
	}

	on_fail = {}

	on_critical_fail = {}

}

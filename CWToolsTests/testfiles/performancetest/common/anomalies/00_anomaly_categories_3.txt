	desc = "TROP_BIOLOGY_DESC"
	picture = GFX_evt_alien_nature
	level = 2
	#risk = 20

	on_spawn = {
	}

	spawn_chance = {
		modifier = {
			add = 3
			is_planet_class = pc_tropical
		}
	}
	#Supposed to be empty, handled in anomaly_events.txt
	on_success = {
	}
	on_fail = {
		if = {
			limit = { owner = { is_borg_empire = no is_assimilator = no } }
			ship_event = { id = anomaly_failure.4000 }
		}
		if = {
			limit = { owner = { is_borg_empire = yes is_assimilator = yes } }
			ship_event = { id = STH_borg_anomaly_failure.3 }
		}
	}
}

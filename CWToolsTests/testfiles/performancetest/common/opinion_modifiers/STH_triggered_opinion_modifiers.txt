			OR = {
				has_country_flag = cardassian_union

			}
			OR = {
				FROM = { has_country_flag = cardassian_union  }
				FROM = { has_country_flag = breen_confederacy  }
				FROM = { has_country_flag = tholian_assembly  }
			}
		}
	}
	opinion = { base = 25 }
}

# Pictish province converts to Scottish
province_event = {
	id = CM.1
	desc = EVTDESC_CM_1
	picture = GFX_evt_stone_church

	trigger = {
		culture = pictish
		owner = {
			NOT = {	culture = pictish }
		}
		NOT = {
			owner = { 
				top_liege = { 
					culture = pictish 
					OR = {
						tier = emperor
						num_of_king_titles = 2
					}
				} 
			}
		}
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTA_CM_1
		culture = scottish
	}
}

# Old Saxon or Old Frankish or Frisian province converts to Dutch
province_event = {
	id = CM.3
	desc = EVTDESC_CM_3
	picture = GFX_evt_tribal_lands

	trigger = {
		county = { region = custom_frisia }
		OR = {
			culture = old_frankish
			culture = frisian
			AND = {
				culture = old_saxon
				has_global_flag = widukind_spawned
				NOT = {
					owner = { 
						top_liege = { 
							culture = old_saxon 
							OR = {
								tier = emperor
								num_of_king_titles = 2
							}
						} 
					}
				}
			}
		}
		NOT = {
			owner = { 
				culture = old_saxon 
				religion_head = { culture = old_saxon }
			}
		}
		owner = {
			NOT = { culture = ROOT }
		}
	}
	
	mean_time_to_happen = {
		months = 3600
		modifier = {
			factor = 0.5
			owner = {
				culture = dutch
			}
		}
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTA_CM_3
		culture = dutch
	}
}

# East Slavic province becomes Russian
province_event = {
	id = CM.11
	desc = EVTDESC_CM_11
	picture = GFX_evt_tribal_lands

	trigger = {
		culture_group = east_slavic
		NOT = { culture = russian }
		owner = {
			OR = {
				culture = russian
				AND = {
					NOT = { culture_group = east_slavic }
					NOT = { culture_group = finno_ugric }
				}
				NOT = { religion_group = pagan_group }
				AND = {
					religion_group = pagan_group
					is_reformed_religion = yes
					NOT = {
						religion_head = { culture = THIS }
					}
				}
			}
		}
		any_province_holding = {
			NOT = { holding_type = nomad }
		}
	}
	
	mean_time_to_happen = {
		months = 1200
		modifier = {
			factor = 0.3
			OR = {
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots
				}
				has_game_rule = {
					name = culture_conversion
					value = faster_melting_pots_and_slower_normal
				}
			}
		}
	}

	option = {
		name = EVTOPTA_CM_11
		culture = russian
	}

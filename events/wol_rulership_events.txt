# Rulership Level Up
character_event = {
	id = WoL.12500
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		if = {
			limit = { 
				OR = {
					trait = indulgent_wastrel
					trait = thrifty_clerk
					trait = fortune_builder
				}
				random = 85
			}
			character_event = { id = WoL.12501 }
		}
		if = {
			limit = {
				NOT = {
					AND = {
						NOT = { religion = jain }
						trait = administrator
					}
				}
				NOT = { has_character_modifier = wol_raja_dharma }
				random = 85
			}
			character_event = { id = WoL.12502 }
		}
		if = {
			limit = {
				NOT = { trait = just }
				random = 85
			}
			character_event = { id = WoL.12503 }
		}
		if = {
			limit = {
				NOT = { trait = diligent }
				random = 85
			}
			character_event = { id = WoL.12504 }
		}
		if = {
			limit = {
				NOT = { trait = ambitious }
				random = 85
			}
			character_event = { id = WoL.12505 }
		}
	}
}

		}
	}
}

		}
	}
